  - name: Checkout dev branch
  uses: actions/checkout@v4
  with:
  ref: dev
  fetch-depth: 0
  
  - name: Set up Node.js
  uses: actions/setup-node@v4
  with:
  node-version: '20'
  
  # 1️⃣ Switch to main and merge dev
  - name: Merge dev → main
  run: |
  git fetch origin main
  git checkout main
  git merge origin/dev --no-edit || echo "✅ No merge conflicts"
  git push origin main
